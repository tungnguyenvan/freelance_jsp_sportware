use sportware;
create table produces(
	id int IDENTITY(1,1) primary key,
	name varchar(255),
	description text,
	price int,
	image text
);

create table users(
	id int IDENTITY(1,1) primary key,
	name varchar(255),
	email varchar(255),
	password varchar(255)
);

create table carts(
	id int IDENTITY(1,1) primary key,
	id_user int,
	id_produce int,

	constraint fk_user foreign key (id_user) references users (id),
	constraint fk_produce foreign key (id_produce) references produces (id)
)
